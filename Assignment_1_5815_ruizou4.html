def generate_sim_data(sim_params):

    p = sim_params.p
    s = sim_params.s
    n = sim_params.n
    N = sim_params.N
    m1 = sim_params.m1
    m0 = sim_params.m0
    csize = sim_params.csize
    
    id1 = np.random.randint(csize, size = n)
    id0 = np.random.randint(csize, size = n)

    Xtrain = np.random.normal(size = (2 * n, p)) * s \
                + np.concatenate([m1[id1,:], m0[id0,:]])
    Ytrain = np.concatenate(([1]*n, [0]*n))

    id1 = np.random.randint(csize, size = N)
    id0 = np.random.randint(csize, size = N)
    Xtest = np.random.normal(size = (2 * N, p)) * s \
                + np.concatenate([m1[id1,:], m0[id0,:]])
    Ytest = np.concatenate(([1]*N, [0]*N))

    return Xtrain, Ytrain, Xtest, Ytest

The following is my_knn implementation: we first calculate the euclidean distance (squared) between two vectors, then sort the distance and get the k closest distance training data and give predictions based on that.

In [13]:
# Euclidean Distance (squared: as we care only the relative ranking here)
# Use vector expression as pointed out by Peng Xu on CampusWire:
def euclidean(x, y):
    p = np.shape(x)[1]              # dimension of each vector, it's 2 here
    XX = np.multiply(x, x)
    YY = np.multiply(y, y)          # element-wise multiplication for x and y
    XY = np.dot(x, y.transpose())   # dot product here
    dist = np.add.outer(np.dot(XX, np.ones(p)), np.dot(YY, np.ones(p))) - 2*XY  # use outer sum
    return dist

def my_knn(Xtrain, Xtest, Ytrain, k):
    # calculate the distance between xtest and xtrain:
    distance = euclidean(Xtrain, Xtest)              # a n*N matrix 
    indices = np.argsort(distance.transpose())       # sort the distance, get indices
                                                     #### using python's argsort, it gives the first index first if two or more values are equal 
    indices_k = indices.transpose()[:k, :]           # first K smallest distance
    
    N = np.shape(Xtest)[0]                           # number of test data
    y_knn = np.zeros((k, N))
    votes = [0]*N
    y_pred = [0]*N
    
    for j in range(N): 
        y_knn[:, j] = Ytrain[indices_k[:, j]]       # y for k-closest training data
        votes[j] = sum(y_knn[:, j])
        if votes[j] > k/2:
            y_pred[j] = 1
        elif votes[j] < k/2:
            y_pred[j] = 0
        else:    # voting tie, resort to (K+1)-th distance for result -- seems more reasonable than a random assignment:
                 # alternatively, delete the K-th element, rely on (K-1) nearest elements only
            y_pred[j] = Ytrain[indices[j, k]]

    return np.array(y_pred)

Distance ties: It's possible that two or more training data have the same distances to a given test data. In such cases, np.argsort will sort the indices from low to high. So when there are distance ties, the smaller index will be chosen. Note that when the order of indices in the dataset has a meaning, e.g., men are put in the first half of the data and women are put in the second half of the data, the algorithm will always choose men since their indices will be smaller when there are distance ties.

Voting ties: When k is an even number, it's possible that the top 2 clusters have the same number of data points, thus a voting tie. In this case, we come up with two possible approaches:

  1. bring in the (k+1)-th distance to break the tie, this essentially brings in more information, and the resulting errors are usually smaller than that of the embedded KNN function.
  2. discard the k-th distance and rely on the previous (k-1) distances only for prediction (i.e., the prediction is the opposite of that of the k-th value), this is equivalent to giving more weights on the smaller distances.

In the report, we use the first approach and bring in the additional (k+1)-th element to break the tie.

Part II: cv-KNN¶

The following function cvKNN calculates the optimal K value for 10-fold cross-validation KNN.

Non-uniqueness of optimal K values in cvKNN:

In case that the resulting optimal K values are not unique (i.e., multiple K values give the (same) smallest CV error), we pick the largest K value among them, to give simpler models and smoother classification boundary.

In [22]:
def cvKNN(traindata, Ytrain, num_folds, plot = True):
    n = len(Ytrain)
    fold_size = int(n / num_folds)
    kvector = range(1, n + 1 - fold_size)             # candidate K values
    k_cverr = np.zeros_like(kvector)

    ########################################
    # Your Code
    # Compute CV errors and store them in k_cverr
    ## split all data into num_folds data samples:
    indices = np.arange(n)
    np.random.shuffle(indices)
    index_sets = np.array_split(indices, num_folds)

    for k in kvector:
        for ifold in range(num_folds):
            train_indices = np.delete(index_sets, obj=ifold, axis=0).ravel()
            _traindata = traindata[train_indices]
            _Ytrain = Ytrain[train_indices]

            test_indices = index_sets[ifold]
            _testdata = traindata[test_indices]
            _Ytest = Ytrain[test_indices]

            knn_model = knn(n_neighbors = k)
            knn_model.fit(_traindata, _Ytrain)
            predict_Y = knn_model.predict(_testdata)
            k_cverr[k-1] += sum(predict_Y != _Ytest)    
    
    k_cverr = k_cverr/n
    
    if plot:
        plt.figure()
        plt.scatter(kvector, k_cverr, color = 'blue', alpha = 0.2, label = 'k_cv_error')
        plt.xlabel("K Values")
        plt.ylabel("CV Errors")
        plt.legend()
        plt.show()
    
    #########################################
    
    tmp = np.column_stack((kvector, k_cverr))
    best_k = tmp[tmp[:,1] == np.min(tmp[:,1])][:,0]
    return int(np.max(best_k))   #### choose maximum k if it is a tie

def bayesPredict(x, sim_params):
    s = sim_params.s
    m1 = sim_params.m1
    m0 = sim_params.m0
    
    d1 = sum(np.exp(- ((m1[i, 0] - x[:, 0]) ** 2 + (m1[i, 1] - x[:, 1]) ** 2) / (2 * s ** 2)) for i in range(len(m1)))
    d0 = sum(np.exp(- ((m0[i, 0] - x[:, 0]) ** 2 + (m0[i, 1] - x[:, 1]) ** 2) / (2 * s ** 2)) for i in range(len(m0)))
    
    y_pred = (d1 >= d0)
    
    return y_pred

def dataSimulation(M, sim_params):
    error_1 = [0]*M
    best_k = [0]*M
    error_best_k = [0]*M
    error_bayes = [0]*M

    for i in range(M):
        Xtrain, Ytrain, Xtest, Ytest = generate_sim_data(sim_params)

        # KNN classification with K=1:
        knn_1 = knn(n_neighbors = 1)
        knn_1.fit(Xtrain, Ytrain)
        test_pred = knn_1.predict(Xtest)
        error_1[i] = np.mean(test_pred != Ytest)
        
        # KNN classification with K chosen by 10-fold cross-validation:
        best_k[i] = cvKNN(Xtrain, Ytrain, 10, plot = False)
        cv_KNN = knn(best_k[i])
        cv_KNN.fit(Xtrain, Ytrain)
        test_pred = cv_KNN.predict(Xtest)
        error_best_k[i] = np.mean(test_pred != Ytest)
        
        # the Bayes rule:
        test_pred = bayesPredict(Xtest, sim_params)
        error_bayes[i] = np.mean(test_pred != Ytest)

    return error_1, best_k, error_best_k, error_bayes
